class Record:
  def __init__(self):
    """Иницилазиция класса записи"""
    self.records = []



[документация]
  def AddNewRecord(self, record):
    """
    Модуль добавления новой записи

    Args:
      record: Переменная которая хранит в себе значение новой записи

    Returns:
      (str): Сообщение о успешном добавлении новой записи
    """
    self.records.append(record)
    return f"Успешное добавление новой записи {record}"





[документация]
  def RedactRecord(self, record, new_record):
    """
    Модуль изменения записи

    Args:
      record: Переменная которая хранит в себе старую запись
      new_record: Переменная которая хранит в себе изменённую старую запись

    Returns:
      (str): Сообщение об успешном изменении старой записи
    """
    self.records.remove(record)
    self.records.append(new_record)
    return f"Успешное изменение записи {record} на {new_record}"

  def DeleteRecord(self, record):
    """
    Модуль удаления записи

      record: Переменная которая хранит в себе данные о записи для удаления

    Returns:
      (str): Сообщение об успешном удалении записи
    """
    self.records.remove(record)
    return f"Успешное удаление записи {record}"

  def ImportRecord(self, file):
    """
    Модуль импорта записей из файла

    Args:
      file: Премеенная которая хранит в себе путь и названия файла для получения записи(-ей) из него

    Returns: Сообщение об успешном импорте записи(-ей) из файла
      (str):

    """
    with open(file) as f:
      recording = f.read()
    return f"Успешный импорт записи(-ей) из файла {file}"

  def ExportRecord(self, file, recording):
    """
    Модуль экспорта записей в файл

    Args:
      file: Премеенная которая хранит в себе путь и названия файла для выгрузки записи(-ей) в него
      recording: Переменная которая хранит в себе данные о записях которые нужно выгрузить в файл

    Returns:
      (str): Сообщение об успешном экспорте записи(-ей) в файл
    """
    with open(file) as f:
      f.write(recording)
    return f"Успешный экспорт записи(-ей) в файл {file}"
